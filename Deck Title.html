        static void Main(string[] args)
        {
            var matrixTester = new MatrixTester();

            var firstTimeDiff = MeasureAction(() =>
            {
                matrixTester.TraverseMatrixBasedOnRows();
            });

            var secondTimeDiff = MeasureAction(() =>
            {
                matrixTester.TraverseMatrixBasedOnColumns();
            });

            Console.WriteLine("1. Variant: {0}ms",
                  firstTimeDiff.ToString());
            Console.WriteLine("2. Variant: {0}ms",
                   secondTimeDiff.ToString());
            Console.WriteLine("Difference: {0}ms",
                  (firstTimeDiff - secondTimeDiff).ToString());

            Console.ReadKey();
        }

        private static long MeasureAction(Action action)
        {
            var sw = new Stopwatch();
            sw.Start();
            action();
            sw.Stop();

            return sw.ElapsedMilliseconds;
        }
class MatrixTester
 {
        private const int rowSize = 10000;
        private const int colSize = 50;

        private int[,] matrix = new int[rowSize, colSize];

        public void TraverseMatrixBasedOnRows()
        {
            for (int row = 0; row < rowSize; row++)
            {
                for (int col = 0; col < colSize; col++)
                {
                    matrix[row, col] = row + col;
                }
            }
        }

        public void TraverseMatrixBasedOnColumns()
        {
            for (int col = 0; col < colSize; col++)
            {
                for (int row = 0; row < rowSize; row++)
                {
                    matrix[row, col] = row + col;
                }
            }
        }
    }
